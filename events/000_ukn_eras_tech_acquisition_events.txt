country_event = {
	id = ukn_automation_tech_acquisition_events.11
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		last_increased_tech = tech_capital_productivity_1
	}
	
	immediate = {
		if = {
			limit = {
				last_increased_tech = tech_capital_productivity_1
			}
			add_research_option = tech_basic_science_lab_2
			add_research_option = tech_power_hub_1
			add_research_option = tech_food_processing_1
			add_research_option = tech_mineral_purification_1
			add_research_option = tech_alloys_1
		}
		if = {
			limit = {
				last_increased_tech = tech_capital_productivity_1
				country_uses_consumer_goods = yes
			}
			add_research_option = tech_luxuries_1
		}
		if = {
			limit = {
				last_increased_tech = tech_capital_productivity_1
				is_gestalt = no
			}
			add_research_option = tech_space_trading
		}
		if = {
			limit = {
				last_increased_tech = tech_capital_productivity_1
				NOR = {
					has_ethic = ethic_spiritualist
					has_ethic = ethic_fanatic_spiritualist
					has_authority = auth_hive_mind
				}
			}
			add_research_option = tech_planetary_infrastructure_1
		}
		if = {
			limit = {
				last_increased_tech = tech_capital_productivity_1
				OR = {
					has_ethic = ethic_spiritualist
					has_ethic = ethic_fanatic_spiritualist
				}
			}
			add_research_option = tech_holographic_rituals
		}
		if = {
			limit = {
				last_increased_tech = tech_capital_productivity_1
				is_hive_empire = yes
			}
			add_research_option = tech_hive_cluster
		}

}
