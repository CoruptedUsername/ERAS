ring_world_titanic_1 = {
	entity = "construction_platform_entity"
	construction_entity = "construction_platform_entity"
	portrait = "GFX_megastructure_construction_background"
	place_entity_on_planet_plane = yes
	prerequisites = { "giga_tech_ringworld_titanic_2" }

	build_time = @ukn_ringworld_p0_build_time_titanic
	resources = {
		category = ukn_megastructure_site
		cost = {
			influence = @ukn_ringworld_p0_cost_influence_titanic
			unity = @ukn_ringworld_p0_cost_unity_titanic
####
			consumer_goods = @ukn_ringworld_p0_cost_consumer_goods_titanic
			ukn_meta_materials = @ukn_ringworld_p0_cost_ukn_meta_materials_titanic
#			ukn_quasi_matter = @ukn_ringworld_p0_cost_ukn_quasi_matter_titanic
#			ukn_causalitons = @ukn_ringworld_p0_cost_ukn_causalitons_titanic
		}
		upkeep = {
			energy = @ukn_ringworld_p0_upkeep_energy_titanic
		}
	}
#######################

	potential = {
		has_technology = giga_tech_ringworld_titanic_2
		NOT = { has_global_flag = ring_titanic_disabled }
	}

	possible = {
		custom_tooltip = { fail_text = "requires_inside_border"			is_inside_border = from }
		custom_tooltip = { fail_text = "requires_surveyed_system"		giga_system_is_surveyed = yes }
	}

	placement_rules = {
		planet_possible = {
			custom_tooltip = { fail_text = "must_build_around_star"		is_star = yes }
			custom_tooltip = { fail_text = "requires_no_anomaly"		has_anomaly = no }
			custom_tooltip = {
				fail_text = "requires_base_ringworld"
				OR = {
					hidden:solar_system = { has_star_flag = ring_world_built }
					solar_system = { any_system_planet = { is_planet_class = pc_ringworld_habitable } }
				}
			}
			custom_tooltip = {
				fail_text = "already_have_tit_ringworld"
				hidden:solar_system = { NOT = { has_star_flag = ring_world_built_one } }
			}
		}
	}

	ai_weight = {
		weight = 300	# Default = 100

		# Can we afford the upkeep

		# Do we want the production

		# Is this system a good location?

		# Do we lean towards this from federation/ethic/civic?

		# How many of this megastructure can the AI build?
		modifier = { factor = 0.00 from = { has_country_flag = has_recently_built_titanic_ring } }
		modifier = { factor = 0.00	from = { check_variable = { which = completed_t_ringworlds value > 1 } } }
		modifier = { factor = 0.25	from = { check_variable = { which = completed_t_ringworlds value = 1 } } }
		modifier = {
			factor = 0
			from = {
				OR = {
					has_country_flag = giga_started_ringworld
					has_country_flag = giga_started_t_ringworld
					has_country_flag = giga_started_b_ringworld
					has_country_flag = giga_started_g_ringworld
				}
			}
		}
	}

	on_build_complete = {
		save_event_target_as = giga_system
		set_star_flag = ring_world_built_one
		fromfrom.planet = {
			save_event_target_as = giga_planet
			if = { limit = { exists = orbital_station } orbital_station = { dismantle = yes } }
			set_planet_flag = megastructure			# Vanilla uses to check if a megastrucure can be built here
			set_planet_flag = has_megastructure		# Vanilla uses to check if an orbital station can be built here
		}
		from = {
			set_timed_country_flag = { flag = has_recently_built_titanic_ring years = 20 }
			country_event = { id = giga_dialog.2101 }	# Notification
			set_country_flag = giga_started_t_ringworld
		}
		set_variable = { which = t_ring_segments value = 0 }
	}
}

			consumer_goods = @ukn_ringworld_p3_gaia_cost_consumer_goods_titanic
			ukn_meta_materials = @ukn_ringworld_p3_gaia_cost_ukn_meta_materials_titanic
#			ukn_quasi_matter = @ukn_ringworld_p3_gaia_cost_ukn_quasi_matter_titanic
#			ukn_causalitons = @ukn_ringworld_p3_gaia_cost_ukn_causalitons_titanic
		}
		upkeep = {
			energy = @ukn_ringworld_p3_gaia_upkeep_energy_titanic
			ukn_meta_materials = @ukn_ringworld_p3_gaia_upkeep_ukn_meta_materials_titanic
		}
	}
#######################

	potential = {
		NOT = { has_global_flag = ring_titanic_disabled }
		owner = { UKN_CAN_BUILD_GAIA_WORLDS = yes }
	}
	possible = { from = { has_technology = giga_tech_ringworld_titanic_2 } }

	ai_weight = { factor = 10 } # If we can build it, we will build it
	on_build_complete = {
		save_event_target_as = giga_system
		fromfrom.planet = { save_event_target_as = giga_planet }
		from = {
			country_event = { id = giga_dialog.2103 }				# Notification
			set_country_flag = has_built_or_repaired_megastructure	# Backup
			set_country_flag = ukn_has_built_megastructure 	# Triggers Master Builders
			if = {
				limit = {
					NOR = {
						has_global_flag = giga_achievements_disabled
						has_country_flag = giga_achievement_74
					}
				}
				set_country_flag = giga_achievement_74
				set_timed_country_flag = { flag = giga_achievement_74_notification days = 30 }
				giga_achievement_sound = yes
			}
		}

		change_variable = { which = t_ring_segments value = 1 }
		if = {
			limit = { check_variable = { which = t_ring_segments value >= 4 } }
			from = { country_event = { id = giga_dialog.2104 } }	# Fully completed Ringworld
			from = {
				remove_country_flag = giga_started_t_ringworld
				change_variable = { which = completed_t_ringworlds value = 1 }
			}
			set_variable = { which = t_ring_segments value = 0 }
		}

		spawn_planet = {
			class = "pc_ringworld_habitable"
			location = fromfrom
			orbit_angle_offset = 2.95
			size = 20
			init_effect = {
				set_planet_entity = {
					entity = "titanic_gaia_habitable_sector_entity"
					graphical_culture = root.from
				}
				set_graphical_culture = root.from
				set_surveyed = { surveyed = yes surveyor = from }
				set_all_comms_surveyed = yes
				clear_blockers = yes
				save_event_target_as = ring_section
				trigger_megastructure_icon = yes
				set_planet_flag = forbid_guillis_planet_modifiers
				set_planet_flag = megastructure			# Vanilla uses to check if a megastrucure can be built here
				set_planet_flag = has_megastructure		# Vanilla uses to check if an orbital station can be built here
				set_planet_flag = colony_event			# Vanilla uses to prevent unwanted events on planets
				set_planet_flag = giga_ringworld_tit
				add_modifier = { modifier = "giga_titanic_housing" days = -1 }
			}
		}
		ukn_set_habitable_ringworld_section_name = yes
		remove_megastructure = fromfrom
	}
}

	entity = ""
	construction_entity = "titanic_quarter_construction_entity"
	portrait = "GFX_megastructure_construction_background"
	upgrade_from = { ring_world_titanic_ruined_d }
	
	build_time = @ukn_ringworld_restore_build_time_titanic
	resources = {
		category = ukn_megastructure_main
		cost = {
			influence = @ukn_ringworld_restore_cost_influence_titanic
			unity = @ukn_ringworld_restore_cost_unity_titanic
####
			consumer_goods = @ukn_ringworld_restore_cost_consumer_goods_titanic
			ukn_meta_materials = @ukn_ringworld_restore_cost_ukn_meta_materials_titanic
####
			ukn_femtomatter = @ukn_ringworld_restore_cost_ukn_femtomatter_titanic
#			ukn_condensics = @ukn_ringworld_restore_cost_ukn_condensics_titanic
#			ukn_quasi_matter = @ukn_ringworld_restore_cost_ukn_quasi_matter_titanic
#			ukn_causalitons = @ukn_ringworld_restore_cost_ukn_causalitons_titanic
		}
		upkeep = {
			energy = @ukn_ringworld_restore_upkeep_energy_titanic
			ukn_meta_materials = @ukn_ringworld_restore_upkeep_ukn_meta_materials_titanic
		}
	}
#######################

	potential = { NOT = { has_global_flag = ring_titanic_disabled } }
	possible = {
		exists = from
		from = { has_technology = tech_mega_engineering }
		custom_tooltip = { fail_text = "requires_surveyed_system"					giga_system_is_surveyed = yes }
	}

	on_build_start = {
		if = {
			limit = {
				from = {
					NOR = {
						graphical_culture = arthropoid_01
						graphical_culture = avian_01
						graphical_culture = fungoid_01
						graphical_culture = humanoid_01
						graphical_culture = mammalian_01
						graphical_culture = molluscoid_01
						graphical_culture = plantoid_01
						graphical_culture = reptilian_01
						graphical_culture = lithoid_01
						graphical_culture = necroid_01
					}
				}
			}
			fromfrom = { set_graphical_culture = mammalian_01 }
		}
		else = { fromfrom = { set_graphical_culture = root.from } }
	}

	ai_weight = { factor = 10 } # If we can build it, we will build it
	on_build_complete = {
		from = {
			set_country_flag = ukn_has_repaired_megastructure
			set_country_flag = has_built_or_repaired_megastructure	# Backup
		}
		save_event_target_as = giga_system
		fromfrom.planet = { save_event_target_as = giga_planet }

#[+0-DEGREE] Support ("Seam") Section
		spawn_planet = {
			class = "pc_ringworld_seam"
			location = fromfrom
			init_effect = {
				set_name = "Support Section"
				set_planet_entity = {
					entity = "titanic_seam_sector_entity"
					graphical_culture = root.from
				}
				set_surveyed = { surveyed = yes surveyor = from }
				set_all_comms_surveyed = yes
				set_planet_flag = forbid_guillis_planet_modifiers
				set_planet_flag = megastructure			# Vanilla uses to check if a megastrucure can be built here
				set_planet_flag = has_megastructure		# Vanilla uses to check if an orbital station can be built here
				set_planet_flag = giga_ringworld_tit
			}
		}
		remove_megastructure = fromfrom
		
#[+30-DEGREE] Support ("Tech") Section
		spawn_megastructure = {
			name = "Support Section"
			type = "ring_world_titanic_3_tech"
			orbit_angle = 300
			orbit_distance = 90
			owner = from
			graphical_culture = fromfrom
		}
		if = {
			limit = { owner = { UKN_USES_TERRESTRIAL_SECTIONS = yes } }
#[-30-DEGREE] Uninhabitable ("Terrestrial") Section
			spawn_megastructure = {
				name = "Terrestrial Ring Section H"
				type = "ring_world_titanic_3_uninhabitable_default"
				orbit_angle = 240
				orbit_distance = 90
				owner = from
				graphical_culture = fromfrom
			}
		}
		if = {
			limit = { owner = { UKN_USES_DESOLATE_SECTIONS = yes } }
#[-30-DEGREE] Uninhabitable ("Desolate") Section
			spawn_megastructure = {
				name = "Desolate Ring Section H"
				type = "ring_world_titanic_3_uninhabitable_machine"
				orbit_angle = 240
				orbit_distance = 90
				owner = from
				graphical_culture = fromfrom
			}
		}
	}
}
